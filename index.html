Example: In DNA strings, symbols “A” and “T” are complements of each other, as “C” and “G”. You have function with one side of the DNA (string, except for Haskell); you need to get the other complementary side. DNA strand is never empty or there is no DNA at all (again, except for Haskell).


function DNAStrand(dna){
    let compl = '';
    for (let i = 0; i < dna.length; i++){
        switch(dna[i]){
            case 'A':
            compl += 'T';
            break;
            case 'T':
            compl += 'A';
            break;
            case 'C':
            compl += 'G';
            break;
            case 'G':
            compl += 'C';
            break;
            }
        }
    return compl;
}
